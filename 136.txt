class Solution {
    public int singleNumber(int[] nums) {
        int l = nums.length;

        for (int i = 0; i < l; i++) {
            int c = 0;

            for (int a = 0; a < l; a++) {
                if (nums[a] == nums[i]) {
                    c++;
                }
            }

            if (c == 1) {
                return nums[i];
            }
        }

        return -1;
    }
}
