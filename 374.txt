public class Solution extends GuessGame {
    public int guessNumber(int n) {
        int low = 1, high = n;
        
        while (low <= high) {
            int mid = (low + high ) / 2; 
            
            int res = guess(mid);
            
            if (res == 0) {
                return mid; // correct answer
            } 
            else if (res == -1) {
                high = mid - 1; // go left
            } 
            else {
                low = mid + 1; // go right
            }
        }
        
        return -1; // should not reach here
    }
}
