class Solution {
    public long maxKelements(int[] nums, int k) {
        PriorityQueue<Integer> maxHeap = new PriorityQueue<>((a, b) -> b - a);

        for (int num : nums) {
            maxHeap.add(num);
        }

        long score = 0;

        while (k > 0) {
            int deleted = maxHeap.poll();
            score += deleted;

            maxHeap.add((int) Math.ceil(deleted / 3.0));

            k--;
        }

        return score;
    }
}
